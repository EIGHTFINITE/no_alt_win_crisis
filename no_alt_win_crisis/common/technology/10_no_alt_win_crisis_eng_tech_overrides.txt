			desc = "TECH_UNLOCK_DESTROYER_CONSTRUCTION_DESC"
		}
	}

	modifier = {
		command_limit_add = 10
	}

	weight_modifier = {
		modifier = {
			factor = 0.1
			NOT = { years_passed > 10 }
		}
		modifier = {
			factor = 10

				has_technology = tech_destroyers
			}
		}
		modifier = {
			factor = 2
			years_passed > 15
		}
		modifier = {
			factor = 3
			years_passed > 20
		}
		modifier = {
			factor = 4
			years_passed > 30
		}
		modifier = {
			factor = 1.25
			has_tradition = tr_supremacy_adopt
		}
	}

	ai_weight = {
		factor = @ai_ship_types_factor

	}
}
